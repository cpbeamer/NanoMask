Get the proxy URL:

  kubectl port-forward svc/{{ include "nanomask.fullname" . }} {{ .Values.service.port }}:{{ .Values.service.port }}

  export NODE_PORT=$(kubectl get svc {{ include "nanomask.fullname" . }} -o jsonpath='{.spec.ports[0].nodePort}')
  export NODE_IP=$(kubectl get nodes -o jsonpath='{.items[0].status.addresses[0].address}')
  echo "http://$NODE_IP:$NODE_PORT"

  echo "http://$SERVICE_IP:{{ .Values.service.port }}"
{{- end }}
